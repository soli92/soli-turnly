```typescript
// types
export type RuleId = 'RB-01' | 'RB-02' | ... | 'RB-17';
export type Severity = 'blocking' | 'warning' | 'info';

  ruleId: RuleId;
  severity: Severity;
```

## Esempio implementazione (RB-01)

```typescript
// lib/rules/validateNoOverlap.ts
import { areIntervalsOverlapping } from 'date-fns';

export function validateNoOverlap(
  newStart: Date,
  newEnd: Date,
  existingShifts: Array<{ startDt: Date; endDt: Date; id: string }>
): ValidationResult {
  const overlapping = existingShifts.filter(s =>
    areIntervalsOverlapping(
      { start: newStart, end: newEnd },
      { start: s.startDt, end: s.endDt },
      { inclusive: false }
    )
  );
  if (overlapping.length === 0) return { valid: true, blocking: [], warnings: [] };
  return {
    valid: false,
    blocking: [{
      ruleId: 'RB-01',
      severity: 'blocking',
      message: `Sovrapposizione con turno ${overlapping[0].id}`,
    }],
    warnings: [],
  };
}
```
