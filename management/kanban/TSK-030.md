## Struttura test visual regression

```typescript
// Esempio: tests/visual/sprint2/dashboard.spec.ts
test.describe('Dashboard admin — visual', () => {
  test('desktop light', async ({ page }) => {
    await page.goto('/dashboard');
    await page.waitForLoadState('networkidle');
    await expect(page).toHaveScreenshot('dashboard-desktop-light.png', { maxDiffPixels: 50 });
  });
  test('mobile light', async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('/dashboard');
    await expect(page).toHaveScreenshot('dashboard-mobile-light.png', { maxDiffPixels: 50 });
  });
});
```

## Struttura test A11y

```typescript
// tests/a11y/sprint2/a11y-sprint2.spec.ts
import AxeBuilder from '@axe-core/playwright';

const pages = [
  { name: 'dashboard', path: '/dashboard', role: 'admin' },
  { name: 'matrix', path: '/matrix', role: 'admin' },
  { name: 'employee-calendar', path: '/calendar', role: 'employee' },
  // ... tutte le pagine Sprint 2 e Sprint 3
];

for (const { name, path, role } of pages) {
  test(`a11y ${name} — zero violazioni critical`, async ({ page }) => {
    // login come role appropriato
    await page.goto(path);
    const results = await new AxeBuilder({ page })
      .withTags(['wcag2a', 'wcag2aa', 'wcag21aa', 'wcag22aa'])
      .analyze();
    const criticals = results.violations.filter(v => v.impact === 'critical' || v.impact === 'serious');
    expect(criticals).toHaveLength(0);
  });
}
```
