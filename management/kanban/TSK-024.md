## Scenari chiave

### dashboard.spec.ts

```typescript
test('RF-K CA2: inbox si aggiorna dopo nuova richiesta', async ({ adminPage, employeePage }) => {
  const countBefore = await adminPage.getByTestId('inbox-badge').textContent();
  await employeePage.goto('/requests/new');
  // ... invia richiesta assenza
  await expect(adminPage.getByTestId('inbox-badge')).not.toHaveText(countBefore!, { timeout: 30_000 });
});
```

### absences.spec.ts

```typescript
test('RF-G CA2: conflitti mostrati prima del salvataggio', async ({ adminPage }) => {
  // Setup: turno esistente il 15 gen per Luca
  await adminPage.goto('/admin/absences');
  await adminPage.getByRole('button', { name: 'Nuova assenza' }).click();
  // ... seleziona Luca, 13–17 gen
  await adminPage.getByRole('button', { name: 'Verifica conflitti' }).click();
  await expect(adminPage.getByTestId('conflict-shift-list')).toContainText('15 gen');
});
```

### coverage.spec.ts (T-DOM-07)

```typescript
test('T-DOM-07: fascia sotto-coperta marcata rossa', async ({ adminPage }) => {
  // Setup: fabbisogno 3 Infermieri notte, pianificati 2
  await adminPage.goto('/admin/coverage?tab=monitor');
  const cell = adminPage.getByTestId('coverage-cell-notte-infermiere');
  await expect(cell).toHaveClass(/under-covered/);
  await expect(cell).toContainText('-1');
});
```

### employee-requests.spec.ts (T-REQ-03, T-SEC-08)

```typescript
test('T-REQ-03: scambio richiede accettazione collega prima di admin', async ({ employeePage, colleaguePage, adminPage }) => {
  // Dipendente propone scambio con collega
  // Admin verifica: richiesta in stato 'in_attesa_collega', approve disabilitato
  // Collega accetta → stato cambia → admin può approvare
  await expect(adminPage.getByTestId('approve-button')).toBeDisabled();
  await colleaguePage.goto('/requests');
  await colleaguePage.getByTestId('swap-accept-button').click();
  await expect(adminPage.getByTestId('approve-button')).toBeEnabled({ timeout: 15_000 });
});
```
